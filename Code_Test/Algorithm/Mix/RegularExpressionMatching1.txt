class Solution {
    public boolean isMatch(String s, String p) {
        int s_length = s.length();
        int p_length = p.length();
        int i = 0, j = 0;
        char temp = 0;

        while (i < s_length && j < p_length) {
            if (p.charAt(j) == s.charAt(i)) {
                temp = s.charAt(i); i++; j++;
            } else if (p.charAt(j) == '.') {
                temp = '.'; i++; j++;
            } else if (p.charAt(j) == '*') {
                while ((i < s_length) && ((temp == '.') || (s.charAt(i) == temp))) {
                    i++;
                }
                j++;
            } else {
                temp = 0;
                if (j + 1 < p_length) {
                    if (p.charAt(j + 1) == '*') j += 2;
                    else break;
                } else break;
            }
        }

        return (i >= s_length && j >= p_length);
    }
}

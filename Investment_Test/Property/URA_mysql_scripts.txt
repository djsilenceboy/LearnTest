URA_CONDOTRANS_HISTORY
------------------------------------------------------------
Find average price for same floor range in same project.

SELECT POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, PROPERTY_TYPE, TENURE_YEAR, FLOOR_AREA_LOWER, ROUND(AVG(PRICE)) AS PRICE_AVG
FROM URA_CONDOTRANS_HISTORY
GROUP BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, PROPERTY_TYPE, TENURE_YEAR, FLOOR_AREA_LOWER
ORDER BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, PROPERTY_TYPE, TENURE_YEAR, FLOOR_AREA_LOWER;

URA_CONDORENT_HISTORY
------------------------------------------------------------
Find average yearly rental for same floor range in same project.

SELECT POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, FLOOR_AREA_LOWER, ROUND(AVG(YEARLY_GROSS_RENT)) AS YEARLY_GROSS_RENT_AVG
FROM URA_CONDORENT_HISTORY
GROUP BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, FLOOR_AREA_LOWER
ORDER BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, FLOOR_AREA_LOWER;

URA_CONDOTRANS_HISTORY
------------------------------------------------------------
Find projects with Price-to-Rent Ratio.

SELECT a.POSTAL_DISTRICT, a.PROJECT_NAME, a.STREET_NAME, a.PROPERTY_TYPE, a.TENURE_YEAR, a.FLOOR_AREA_LOWER, a.PRICE_AVG, b.YEARLY_GROSS_RENT_AVG, ROUND(a.PRICE_AVG / b.YEARLY_GROSS_RENT_AVG) AS PRICE_RENT_RATIO
FROM
  (SELECT POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, PROPERTY_TYPE, TENURE_YEAR, FLOOR_AREA_LOWER, ROUND(AVG(PRICE)) AS PRICE_AVG
   FROM URA_CONDOTRANS_HISTORY
   GROUP BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, PROPERTY_TYPE, TENURE_YEAR, FLOOR_AREA_LOWER) a
  INNER JOIN
  (SELECT POSTAL_DISTRICT, PROJECT_NAME, FLOOR_AREA_LOWER, ROUND(AVG(YEARLY_GROSS_RENT)) AS YEARLY_GROSS_RENT_AVG
   FROM URA_CONDORENT_HISTORY
   GROUP BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, FLOOR_AREA_LOWER) b
  ON ((a.POSTAL_DISTRICT = b.POSTAL_DISTRICT)
      AND (a.PROJECT_NAME = b.PROJECT_NAME)
      AND (a.FLOOR_AREA_LOWER = b.FLOOR_AREA_LOWER))
ORDER BY a.FLOOR_AREA_LOWER, a.PRICE_AVG, PRICE_RENT_RATIO;
------------------------------------------------------------
Find projects, which
1. price is <= s$850,000,
2. Price-to-Rent Ratio is <= 30 years.

SELECT a.POSTAL_DISTRICT, a.PROJECT_NAME, a.STREET_NAME, a.PROPERTY_TYPE, a.TENURE_YEAR, a.FLOOR_AREA_LOWER, a.PRICE_AVG, b.YEARLY_GROSS_RENT_AVG, ROUND(a.PRICE_AVG / b.YEARLY_GROSS_RENT_AVG) AS PRICE_RENT_RATIO
FROM
  (SELECT POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, PROPERTY_TYPE, TENURE_YEAR, FLOOR_AREA_LOWER, ROUND(AVG(PRICE)) AS PRICE_AVG
   FROM URA_CONDOTRANS_HISTORY
   GROUP BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, PROPERTY_TYPE, TENURE_YEAR, FLOOR_AREA_LOWER) a
  INNER JOIN
  (SELECT POSTAL_DISTRICT, PROJECT_NAME, FLOOR_AREA_LOWER, ROUND(AVG(YEARLY_GROSS_RENT)) AS YEARLY_GROSS_RENT_AVG
   FROM URA_CONDORENT_HISTORY
   GROUP BY POSTAL_DISTRICT, PROJECT_NAME, STREET_NAME, FLOOR_AREA_LOWER) b
  ON ((a.POSTAL_DISTRICT = b.POSTAL_DISTRICT)
      AND (a.PROJECT_NAME = b.PROJECT_NAME)
      AND (a.FLOOR_AREA_LOWER = b.FLOOR_AREA_LOWER))
WHERE (a.PRICE_AVG <= 850000)
      AND (ROUND(a.PRICE_AVG / b.YEARLY_GROSS_RENT_AVG) <= 30)
ORDER BY a.FLOOR_AREA_LOWER, a.PRICE_AVG, PRICE_RENT_RATIO;

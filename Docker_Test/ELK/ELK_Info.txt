Custom network
------------------------------------------------------------
sudo docker network create dj_test_elk
sudo docker network ls | grep elk
sudo docker network rm dj_test_elk
================================================================================
Elasticsearch
------------------------------------------------------------
sudo docker build -t dj_test_elasticsearch -f Dockerfile_Elasticsearch .
sudo docker create -it --net dj_test_elk -p 9200:9200 -p 9300:9300 --name dj_test_elasticsearch dj_test_elasticsearch
sudo docker start dj_test_elasticsearch
sudo docker exec -it dj_test_elasticsearch bash -l

sudo docker rm -f dj_test_elasticsearch
sudo docker rmi dj_test_elasticsearch
